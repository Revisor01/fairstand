**A) server/src/routes/auth.ts** (neue Datei):

```typescript
import type { FastifyInstance } from 'fastify';
import { z } from 'zod';
import { eq } from 'drizzle-orm';
import { db } from '../db/index.js';
import { shops } from '../db/schema.js';

async function hashPin(pin: string): Promise<string> {
  const encoder = new TextEncoder();
  const data = encoder.encode(pin);
  const hash = await crypto.subtle.digest('SHA-256', data);
  return Array.from(new Uint8Array(hash))
    .map(b => b.toString(16).padStart(2, '0'))
    .join('');
}

const PinAuthSchema = z.object({
  pin: z.string().min(4).max(8),
});

export async function authRoutes(fastify: FastifyInstance) {
  // POST /api/auth/pin — PIN gegen shops-Tabelle prüfen, shopId + Token zurückgeben
  fastify.post('/auth/pin', async (request, reply) => {
    const result = PinAuthSchema.safeParse(request.body);
    if (!result.success) return reply.status(400).send({ error: 'Ungültige Anfrage' });

    const { pin } = result.data;
    const pinHash = await hashPin(pin);

    const shop = db.select().from(shops).where(eq(shops.pin, pinHash)).get();
    if (!shop) return reply.status(401).send({ error: 'Falscher PIN' });

    // Token = einfaches Session-Token (per Discretion: zufälliger String, kein JWT-Overhead)
    // Token wird client-seitig in idb-keyval gespeichert, kein server-seitiges Token-Management
    const token = crypto.randomUUID();

    return reply.send({
      shopId: shop.shopId,
      shopName: shop.name,
      token,
    });
  });
}
```

import { shops, products } from './schema.js';
import { eq } from 'drizzle-orm';

const SHOP_ID = 'st-secundus-hennstedt';
const SHOP_NAME = 'St. Secundus Hennstedt';
const SHOP_PIN = '140381'; // 6-stellig

async function hashPin(pin: string): Promise<string> {
  const encoder = new TextEncoder();
  const data = encoder.encode(pin);
  const hash = await crypto.subtle.digest('SHA-256', data);
  return Array.from(new Uint8Array(hash))
    .map(b => b.toString(16).padStart(2, '0'))
    .join('');
}

// SEED_PRODUCTS: alle 33 Produkte aus client/src/db/seed.ts
// Kategorien: Kunsthandwerk (4), Kaffee (1), Schokolade (6), Süsswaren (15), Gebäck (4), Sonstiges (2)
// Hier die vollständige Liste einfügen (copy aus client/src/db/seed.ts, Typ anpassen)
const SEED_PRODUCTS: Array<{
  articleNumber: string;
  name: string;
  category: string;
  purchasePrice: number;
  salePrice: number;
  vatRate: number;
  stock: number;
  active: boolean;
  minStock: number;
}> = [
  // --- Alle 33 Einträge aus client/src/db/seed.ts kopieren ---
];

export async function ensureShopSeeded(): Promise<void> {
  // Idempotent: Nichts tun wenn Shop bereits existiert
  const existing = db.select().from(shops).where(eq(shops.shopId, SHOP_ID)).get();
  if (existing) return;

  const pinHash = await hashPin(SHOP_PIN);
  const now = Date.now();

  // Shop anlegen
  db.insert(shops).values({
    id: crypto.randomUUID(),
    shopId: SHOP_ID,
    name: SHOP_NAME,
    pin: pinHash,
    createdAt: now,
  }).run();

  // Produkte anlegen (nur wenn noch keine existieren für diesen Shop)
  const productCount = db.select().from(products).where(eq(products.shopId, SHOP_ID)).all().length;
  if (productCount === 0) {
    for (const p of SEED_PRODUCTS) {
      db.insert(products).values({
        id: crypto.randomUUID(),
        shopId: SHOP_ID,
        ...p,
        updatedAt: now,
      }).run();
    }
  }
}
